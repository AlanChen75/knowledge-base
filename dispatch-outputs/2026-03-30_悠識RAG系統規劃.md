### 2.1 檔案掃描器

```python
# scanner.py — NAS 檔案掃描與過濾
import os
import json
from pathlib import Path
from datetime import datetime

SUPPORTED_EXTENSIONS = {
    'text': ['.pages', '.md', '.txt', '.rtf', '.docx'],
    'vision': ['.key', '.keynote'],
    'pdf': ['.pdf'],
    'skip': ['.mp4', '.mov', '.avi', '.mp3', '.wav', '.m4a',
             '.pptx', '.ppt']  # PPTX = 客戶 brief，跳過
}

CLEO_SIGNATURE = '_cleo'  # 已處理標記

def scan_department_folder(base_path: str, department: str) -> list[dict]:
    """掃描部門資料夾，回傳待處理檔案清單"""
    dept_path = Path(base_path) / department
    files_to_process = []

    for file_path in dept_path.rglob('*'):
        if not file_path.is_file():
            continue
        if CLEO_SIGNATURE in file_path.stem:
            continue  # 已處理過，跳過

        ext = file_path.suffix.lower()
        file_type = None
        for ftype, extensions in SUPPORTED_EXTENSIONS.items():
            if ext in extensions:
                file_type = ftype
                break

        if file_type == 'skip' or file_type is None:
            continue

        files_to_process.append({
            'path': str(file_path),
            'name': file_path.name,
            'extension': ext,
            'type': file_type,
            'department': department,
            'size_mb': file_path.stat().st_size / (1024 * 1024),
            'modified': datetime.fromtimestamp(
                file_path.stat().st_mtime
            ).isoformat(),
            'relative_path': str(file_path.relative_to(dept_path))
        })

    return sorted(files_to_process, key=lambda x: x['modified'])


def create_processing_manifest(base_path: str,
                                departments: list[str]) -> dict:
    """建立完整的處理清單"""
    manifest = {
        'created_at': datetime.now().isoformat(),
        'departments': {},
        'total_files': 0,
        'total_size_mb': 0
    }

    for dept in departments:
        files = scan_department_folder(base_path, dept)
        manifest['departments'][dept] = {
            'files': files,
            'count': len(files),
            'size_mb': sum(f['size_mb'] for f in files)
        }
        manifest['total_files'] += len(files)
        manifest['total_size_mb'] += manifest['departments'][dept]['size_mb']

    return manifest


# 使用範例
if __name__ == '__main__':
    NAS_BASE = '/Volumes/NAS/Company'
    DEPARTMENTS = ['UR', 'UXD', 'EDU']

    manifest = create_processing_manifest(NAS_BASE, DEPARTMENTS)
    print(f"待處理：{manifest['total_files']} 檔案，"
          f"共 {manifest['total_size_mb']:.1f} MB")

    # 存成 JSON 供後續管線使用
    with open('processing_manifest.json', 'w') as f:
        json.dump(manifest, f, ensure_ascii=False, indent=2)
```

### 2.2 內容擷取器

```python
# extractors.py — 各格式檔案內容擷取
import subprocess
import base64
import json
from pathlib import Path
from anthropic import Anthropic

client = Anthropic()

# ── Pages 擷取：透過 osascript 直接抓文字 ──
def extract_pages(file_path: str) -> dict:
    """用 AppleScript 擷取 Pages 文字內容"""
    script = f'''
    tell application "Pages"
        open POSIX file "{file_path}"
        delay 2
        set doc to front document
        set docText to body text of doc
        close doc saving no
        return docText
    end tell
    '''
    result = subprocess.run(
        ['osascript', '-e', script],
        capture_output=True, text=True, timeout=60
    )

    if result.returncode != 0:
        raise Exception(f"Pages 擷取失敗: {result.stderr}")

    return {
        'text': result.stdout.strip(),
        'method': 'pages_osascript',
        'source_file': file_path
    }


# ── Keynote 擷取：先匯出圖片，再用 Claude Haiku Vision 理解 ──
def extract_keynote(file_path: str,
                    output_dir: str = '/tmp/keynote_slides') -> dict:
    """匯出 Keynote 每頁為圖片，用 Haiku Vision 理解內容"""
    Path(output_dir).mkdir(parents=True, exist_ok=True)

    # Step 1: 用 AppleScript 匯出每頁為 PNG
    export_script = f'''
    tell application "Keynote"
        open POSIX file "{file_path}"
        delay 3
        set doc to front document
        set slideCount to count of slides of doc
        export doc to POSIX file "{output_dir}" as slide images ¬
            with properties {{image format:PNG, skipped slides:false}}
        close doc saving no
        return slideCount
    end tell
    '''
    result = subprocess.run(
        ['osascript', '-e', export_script],
        capture_output=True, text=True, timeout=120
    )
    slide_count = int(result.stdout.strip()) if result.stdout.strip() else 0

    # Step 2: 用 Claude Haiku Vision 理解每張 slide
    slides_content = []
    slide_images = sorted(Path(output_dir).glob('*.png'))

    for i, img_path in enumerate(slide_images):
        with open(img_path, 'rb') as f:
            img_b64 = base64.standard_b64encode(f.read()).decode()

        response = client.messages.create(
            model="claude-haiku-4-5-20251001",
            max_tokens=2000,
            messages=[{
                "role": "user",
                "content": [
                    {
                        "type": "image",
                        "source": {
                            "type": "base64",
                            "media_type": "image/png",
                            "data": img_b64
                        }
                    },
                    {
                        "type": "text",
                        "text": (
                            "請詳細描述這張投影片的所有內容，包括："
                            "1. 標題和副標題\n"
                            "2. 所有文字內容\n"
                            "3. 圖表/流程圖的結構和資訊\n"
                            "4. 表格數據\n"
                            "5. 視覺元素的意義\n"
                            "用繁體中文輸出，保留原始術語。"
                        )
                    }
                ]
            }]
        )

        slides_content.append({
            'slide_number': i + 1,
            'text': response.content[0].text,
            'image_path': str(img_path)
        })

    return {
        'slides': slides_content,
        'full_text': '\n\n---\n\n'.join(
            f"[Slide {s['slide_number']}]\n{s['text']}"
            for s in slides_content
        ),
        'slide_count': len(slides_content),
        'method': 'keynote_haiku_vision',
        'source_file': file_path
    }


# ── PDF 擷取 ──
def extract_pdf(file_path: str) -> dict:
    """擷取 PDF 文字，含 fallback OCR"""
    import fitz  # PyMuPDF

    doc = fitz.open(file_path)
    pages_text = []

    for page_num, page in enumerate(doc):
        text = page.get_text()
        if len(text.strip()) < 50:
            # 文字太少，可能是掃描件，用 Vision 處理
            pix = page.get_pixmap(dpi=200)
            img_b64 = base64.standard_b64encode(
                pix.tobytes("png")
            ).decode()

            response = client.messages.create(
                model="claude-haiku-4-5-20251001",
                max_tokens=2000,
                messages=[{
                    "role": "user",
                    "content": [
                        {
                            "type": "image",
                            "source": {
                                "type": "base64",
                                "media_type": "image/png",
                                "data": img_b64
                            }
                        },
                        {
                            "type": "text",
                            "text": "請完整轉錄這頁文件的所有文字內容。"
                        }
                    ]
                }]
            )
            text = response.content[0].text

        pages_text.append({
            'page': page_num + 1,
            'text': text
        })

    doc.close()

    return {
        'pages': pages_text,
        'full_text': '\n\n'.join(p['text'] for p in pages_text),
        'page_count': len(pages_text),
        'method': 'pdf_pymupdf',
        'source_file': file_path
    }


# ── DOCX 擷取 ──
def extract_docx(file_path: str) -> dict:
    """擷取 Word 文件文字"""
    from docx import Document

    doc = Document(file_path)
    paragraphs = [p.text for p in doc.paragraphs if p.text.strip()]

    return {
        'text': '\n\n'.join(paragraphs),
        'method': 'docx_python',
        'source_file': file_path
    }


# ── 統一入口 ──
EXTRACTORS = {
    'text': extract_pages,      # .pages
    'vision': extract_keynote,  # .key, .keynote
    'pdf': extract_pdf,
    'docx': extract_docx
}

def extract_content(file_info: dict) -> dict:
    """根據檔案類型選擇對應擷取器"""
    ext = file_info['extension']

    if ext == '.docx':
        return extract_docx(file_info['path'])
    elif ext == '.pdf':
        return extract_pdf(file_info['path'])

    extractor = EXTRACTORS.get(file_info['type'])
    if not extractor:
        raise ValueError(f"不支援的檔案類型: {ext}")
    return extractor(file_info['path'])
```

### 2.3 Chunking 策略

```python
# chunker.py — 語義感知的文本切割
from dataclasses import dataclass

@dataclass
class Chunk:
    text: str
    chunk_index: int
    start_char: int
    end_char: int
    metadata: dict

def semantic_chunk(text: str,
                   max_chars: int = 1500,
                   overlap_chars: int = 200,
                   source_metadata: dict = None) -> list[Chunk]:
    """
    語義感知切割策略：
    1. 優先在段落邊界切割
    2. 其次在句子邊界切割
    3. 保留前後 overlap 確保上下文連貫
    """
    if not text or not text.strip():
        return []

    # 先按段落分割
    paragraphs = [p.strip() for p in text.split('\n\n') if p.strip()]

    chunks = []
    current_text = ""
    current_start = 0
    char_pos = 0

    for para in paragraphs:
        # 如果加上這段會超過上限
        if len(current_text) + len(para) + 2 > max_chars:
            if current_text:
                chunks.append(Chunk(
                    text=current_text.strip(),
                    chunk_index=len(chunks),
                    start_char=current_start,
                    end_char=char_pos,
                    metadata=source_metadata or {}
                ))
                # overlap: 保留最後 overlap_chars
                overlap_text = current_text[-overlap_chars:]
                current_text = overlap_text + '\n\n' + para
                current_start = char_pos - overlap_chars
            else:
                current_text = para
        else:
            if current_text:
                current_text += '\n\n' + para
            else:
                current_text = para
                current_start = char_pos

        char_pos += len(para) + 2

    # 最後一段
    if current_text.strip():
        chunks.append(Chunk(
            text=current_text.strip(),
            chunk_index=len(chunks),
            start_char=current_start,
            end_char=char_pos,
            metadata=source_metadata or {}
        ))

    return chunks


# Keynote 特殊策略：每 2-3 張 slide 為一個 chunk
def chunk_keynote_slides(slides: list[dict],
                         slides_per_chunk: int = 3,
                         source_metadata: dict = None) -> list[Chunk]:
    """Keynote 依 slide 分組切割"""
    chunks = []

    for i in range(0, len(slides), slides_per_chunk):
        group = slides[i:i + slides_per_chunk]
        text = '\n\n---\n\n'.join(
            f"[Slide {s['slide_number']}]\n{s['text']}"
            for s in group
        )
        meta = {
            **(source_metadata or {}),
            'slide_range': f"{group[0]['slide_number']}-"
                          f"{group[-1]['slide_number']}"
        }
        chunks.append(Chunk(
            text=text,
            chunk_index=len(chunks),
            start_char=0,
            end_char=len(text),
            metadata=meta
        ))

    return chunks
```

### 2.4 批次處理器

```python
# batch_processor.py — 批次處理主程式
import json
import time
import logging
from pathlib import Path
from datetime import datetime

from scanner import create_processing_manifest, CLEO_SIGNATURE
from extractors import extract_content
from chunker import semantic_chunk, chunk_keynote_slides
from qdrant_ops import upsert_chunks, get_embedding

logging.basicConfig(level=logging.INFO,
                    format='%(asctime)s [%(levelname)s] %(message)s')
logger = logging.getLogger(__name__)

PROCESSING_LOG = 'processing_log.jsonl'

def process_single_file(file_info: dict) -> dict:
    """處理單一檔案：擷取 → 切割 → 寫入向量資料庫"""
    logger.info(f"處理: {file_info['name']}")
    result = {'file': file_info['path'], 'status': 'success'}

    try:
        # 1. 擷取內容
        content = extract_content(file_info)

        # 2. 準備 metadata
        metadata = {
            'source_file': file_info['name'],
            'department': file_info['department'],
            'file_type': file_info['extension'],
            'modified_date': file_info['modified'],
            'relative_path': file_info['relative_path'],
            'extraction_method': content['method'],
            'processed_at': datetime.now().isoformat()
        }

        # 3. 切割
        if file_info['type'] == 'vision' and 'slides' in content:
            chunks = chunk_keynote_slides(
                content['slides'],
                source_metadata=metadata
            )
            # 保存截圖路徑到 metadata
            for chunk in chunks:
                chunk.metadata['has_slides'] = True
        else:
            chunks = semantic_chunk(
                content.get('full_text', content.get('text', '')),
                source_metadata=metadata
            )

        # 4. 生成 embedding 並寫入 Qdrant
        for chunk in chunks:
            embedding = get_embedding(chunk.text)
            upsert_chunks(
                collection=f"{file_info['department'].lower()}_docs",
                chunk=chunk,
                embedding=embedding
            )

        # 5. 重新命名標記已處理
        original = Path(file_info['path'])
        new_name = f"{original.stem}{CLEO_SIGNATURE}{original.suffix}"
        new_path = original.parent / new_name
        original.rename(new_path)

        result['chunks_created'] = len(chunks)
        logger.info(f"  完成: {len(chunks)} chunks")

    except Exception as e:
        result['status'] = 'error'
        result['error'] = str(e)
        logger.error(f"  失敗: {e}")

    return result


def batch_process(manifest_path: str,
                  department: str = None,
                  rate_limit_delay: float = 0.5):
    """批次處理整份 manifest"""
    with open(manifest_path) as f:
        manifest = json.load(f)

    departments = ([department] if department
                   else list(manifest['departments'].keys()))

    for dept in departments:
        dept_data = manifest['departments'].get(dept, {})
        files = dept_data.get('files', [])
        logger.info(f"\n{'='*50}")
        logger.info(f"部門: {dept} | 檔案數: {len(files)}")
        logger.info(f"{'='*50}")

        for i, file_info in enumerate(files):
            logger.info(f"\n[{i+1}/{len(files)}]")
            result = process_single_file(file_info)

            # 寫入處理紀錄
            with open(PROCESSING_LOG, 'a') as log:
                log.write(json.dumps(result, ensure_ascii=False) + '\n')

            # API rate limit 控制
            time.sleep(rate_limit_delay)


if __name__ == '__main__':
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument('--manifest', default='processing_manifest.json')
    parser.add_argument('--department', default=None)
    parser.add_argument('--delay', type=float, default=0.5)
    args = parser.parse_args()

    batch_process(args.manifest, args.department, args.delay)
```

```python
from qdrant_client.models import (

# ── 初始化 ──
qdrant = QdrantClient(host="localhost", port=6333)

def init_collection(name: str):

    collections = [c.name for c in qdrant.get_collections().collections]
    if name not in collections:

# ── 寫入 ──
def upsert_chunks(collection: str, chunk, embedding: list[float]):
    """將 chunk 寫入 Qdrant"""
    # 用內容 hash 作為去重依據
    content_hash = hashlib.md5(chunk.text.encode()).hexdigest()
    point_id = str(uuid.uuid5(uuid.NAMESPACE_DNS, content_hash))

    qdrant.upsert(
        collection_name=collection,
        points=[
            PointStruct(
                id=point_id,
                vector=embedding,
                payload={
                    "text": chunk.text,
                    "chunk_index": chunk.chunk_index,
                    **chunk.metadata
                }
            )
        ]
    )


# ── 搜尋 ──
def search(collection: str,
           query_embedding: list[float],
           department: str = None,
           file_type: str = None,
           limit: int = 10) -> list:
    """向量搜尋 + metadata 過濾"""
    filters = []
    if department:
        filters.append(
            FieldCondition(
                key="department",
                match=MatchValue(value=department)
            )
        )
    if file_type:
        filters.append(
            FieldCondition(
                key="file_type",
                match=MatchValue(value=file_type)
            )
        )

    search_filter = Filter(must=filters) if filters else None

    results = qdrant.search(
        collection_name=collection,
        query_vector=query_embedding,
        query_filter=search_filter,
        limit=limit,
        with_payload=True
    )

    return results


# 初始化所有 collections
if __name__ == '__main__':
    for dept in ['ur_docs', 'uxd_docs', 'edu_docs', 'shared_docs']:
        init_collection(dept)
    print("所有 collections 初始化完成")
```

### 3.2 Metadata 結構

| 欄位 | 類型 | 說明 |
|------|------|------|
| `text` | string | chunk 原始文字 |
| `chunk_index` | int | 在原文件中的位置 |
| `source_file` | keyword | 來源檔名 |
| `department` | keyword | 部門：UR/UXD/EDU |
| `file_type` | keyword | .pages/.key/.pdf/.docx |
| `modified_date` | keyword | 檔案最後修改日期 |
| `relative_path` | string | NAS 上的相對路徑 |
| `extraction_method` | keyword | 擷取方式 |
| `processed_at` | string | 處理時間 |
| `has_slides` | bool | 是否有投影片截圖 |

## 四、查詢處理管線（核心重點）

這是你提到最需要打磨的部分。設計為三步驟流程：

### 4.1 Step 1: 查詢意圖分析（Haiku）

```python
# query_pipeline.py
from anthropic import Anthropic

client = Anthropic()

INTENT_ANALYSIS_PROMPT = """你是悠識數位的內部知識搜尋助手。
公司有三個部門的知識庫：
- UR（使用者研究）：易用性測試報告、訪談紀錄、研究方法論、
  persona、journey map、競品分析
- UXD（體驗設計）：設計規範、wireframe 說明、UI pattern、
  設計提案、客戶專案交付物
- EDU（教育訓練）：課程教材、工作坊紀錄、培訓計畫、對外分享

請分析使用者的搜尋意圖，輸出 JSON：
{
  "rephrased_query": "改寫為更精確的搜尋語句（繁體中文）",
  "english_query": "英文版搜尋語句（部分文件可能含英文）",
  "keywords": ["關鍵詞1", "關鍵詞2"],
  "target_departments": ["UR"],  // 可能相關的部門，可多選
  "intent_type": "factual|exploratory|comparative",
  "time_relevance": "recent|any"  // 是否需要最新資料
}

只輸出 JSON，不要其他文字。"""


def analyze_query(user_query: str,
                  selected_department: str = None) -> dict:
    """用 Haiku 分析查詢意圖"""
    context = f"使用者選擇的部門篩選：{selected_department}" \
              if selected_department else "使用者未指定部門"

    response = client.messages.create(
        model="claude-haiku-4-5-20251001",
        max_tokens=500,
        system=INTENT_ANALYSIS_PROMPT,
        messages=[{
            "role": "user",
            "content": f"{context}\n\n搜尋內容：{user_query}"
        }]
    )

    import json
    return json.loads(response.content[0].text)
```

### 4.2 Step 2: 混合搜尋 + RRF 排序

```python
def hybrid_search(intent: dict, limit: int = 20) -> list:
    """
    混合搜尋策略：
    1. 向量搜尋（語義相似度）
    2. 關鍵字搜尋（精確匹配）
    3. RRF (Reciprocal Rank Fusion) 合併排序
    """
    results_by_id = {}

    # 向量搜尋 — 中文 query
    zh_embedding = get_embedding(intent['rephrased_query'])
    for dept in intent['target_departments']:
        collection = f"{dept.lower()}_docs"
        vector_results = search(
            collection, zh_embedding, limit=limit
        )
        for rank, hit in enumerate(vector_results):
            pid = hit.id
            if pid not in results_by_id:
                results_by_id[pid] = {
                    'payload': hit.payload, 'rrf_score': 0
                }
            results_by_id[pid]['rrf_score'] += 1.0 / (60 + rank)

    # 向量搜尋 — 英文 query（捕捉英文文件）
    if intent.get('english_query'):
        en_embedding = get_embedding(intent['english_query'])
        for dept in intent['target_departments']:
            collection = f"{dept.lower()}_docs"
            en_results = search(
                collection, en_embedding, limit=limit // 2
            )
            for rank, hit in enumerate(en_results):
                pid = hit.id
                if pid not in results_by_id:
                    results_by_id[pid] = {
                        'payload': hit.payload, 'rrf_score': 0
                    }
                results_by_id[pid]['rrf_score'] += 1.0 / (60 + rank)

    # 按 RRF score 排序
    sorted_results = sorted(
        results_by_id.items(),
        key=lambda x: x[1]['rrf_score'],
        reverse=True
    )

    return [
        {'id': rid, **rdata}
        for rid, rdata in sorted_results[:limit]
    ]
```

### 4.3 Step 3: 結果重排 + 生成回答（Sonnet）

```python
RERANK_PROMPT = """你是悠識數位的知識管理助手。
根據使用者的問題和檢索到的知識片段，請：

1. 判斷每個片段與問題的相關程度（高/中/低）
2. 用這些知識回答使用者的問題
3. 標註資訊來源（檔案名稱和部門）

如果知識片段不足以回答，請明確說明缺少什麼資訊。
不要編造不在知識片段中的內容。"""


def rerank_and_answer(user_query: str,
                      search_results: list) -> dict:
    """用 Sonnet 重排結果並生成回答"""
    # 組裝 context
    context_parts = []
    for i, result in enumerate(search_results[:10]):
        payload = result['payload']
        context_parts.append(
            f"[片段 {i+1}] "
            f"來源: {payload['source_file']} "
            f"({payload['department']})\n"
            f"{payload['text'][:1000]}"
        )

    context = "\n\n---\n\n".join(context_parts)

    response = client.messages.create(
        model="claude-sonnet-4-5-20241022",
        max_tokens=2000,
        system=RERANK_PROMPT,
        messages=[{
            "role": "user",
            "content": (
                f"問題：{user_query}\n\n"
                f"檢索到的知識片段：\n\n{context}"
            )
        }]
    )

    return {
        'answer': response.content[0].text,
        'sources': [
            {
                'file': r['payload']['source_file'],
                'department': r['payload']['department'],
                'relevance_score': r['rrf_score'],
                'text_preview': r['payload']['text'][:200]
            }
            for r in search_results[:10]
        ]
    }
```

### 5.1 後端 API（FastAPI）

```python
# api.py
from fastapi import FastAPI, HTTPException
from fastapi.staticfiles import StaticFiles
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from query_pipeline import analyze_query, hybrid_search, rerank_and_answer

app = FastAPI(title="悠識知識檢索系統")

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

class SearchRequest(BaseModel):
    query: str
    department: str | None = None  # UR / UXD / EDU / None=全部

class SearchResponse(BaseModel):
    answer: str
    sources: list[dict]
    intent: dict

@app.post("/api/search", response_model=SearchResponse)
async def search_knowledge(req: SearchRequest):
    try:
        # Step 1: 意圖分析
        intent = analyze_query(req.query, req.department)

        # 如果使用者指定了部門，覆蓋 AI 判斷
        if req.department:
            intent['target_departments'] = [req.department]

        # Step 2: 混合搜尋
        results = hybrid_search(intent)

        if not results:
            return SearchResponse(
                answer="沒有找到相關的知識內容。"
                       "試試用不同的關鍵字或擴大部門範圍？",
                sources=[],
                intent=intent
            )

        # Step 3: 重排 + 回答
        response = rerank_and_answer(req.query, results)

        return SearchResponse(
            answer=response['answer'],
            sources=response['sources'],
            intent=intent
        )

    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.get("/api/stats")
```

## 七、成本估算

以 UR 部門為例，假設 500 份文件：

| 項目 | 單價 | 估算用量 | 月成本 |
|------|------|----------|--------|
| **一次性處理** | | | |
| Haiku Vision（Keynote） | $0.80/1M input | ~200 份 × 20 slides | ~$3 |
| Embedding（Voyage-3） | $0.06/1M tokens | ~500 份 × 5 chunks | ~$0.5 |
| **日常查詢** | | | |
| Haiku（意圖分析） | $0.80/1M input | 50 次/天 × 22 天 | ~$1 |
| Sonnet（重排回答） | $3/1M input | 50 次/天 × 22 天 | ~$5 |
| Embedding（查詢） | $0.06/1M tokens | 50 次/天 × 22 天 | ~$0.1 |
| **Qdrant** | 自架 | Docker | $0 |
| | | **月度總計** | **~$6-10** |

擴展到全公司三個部門，大約 $15-25/月。非常合理。

### 降成本備案

- 只在需要生成回答時才呼叫 Sonnet

## 十、關鍵建議

1. **查詢分析的 prompt 要持續迭代。** 第一版一定不夠好。建議每週看一次查詢 log，把失敗案例加進 prompt 的 few-shot examples。

2. **Chunking 大小很關鍵。** 太大會稀釋相關性，太小會失去上下文。建議從 1500 字元開始，根據搜尋品質調整。

3. **不要急著做全文搜尋。** 先把向量搜尋做好，確認語義搜尋的品質，再加關鍵字搜尋做補充。

4. **Keynote Vision 的成本可以優化。** 如果很多 Keynote 的 slide 是純文字，可以先用 OCR 試，只有圖表才送 Haiku Vision。

5. **Claude Pro 的額度問題。** 開發階段建議直接用 API key，費用透明可控，不會有額度焦慮。以上面的成本估算，API 比 Pro 訂閱便宜很多。
